# Clone este repositório
$ git clone <https://github.com/Samuelbrandani/marvel-api>

# Acesse a pasta do projeto no terminal/cmd
$ cd marvel-api

# Instale as dependências
$ npm install

# Rode os testes
$ npm test

# Execute a aplicação
$ npm start

# O servidor inciará na porta:8081 - acesse <http://localhost:8081>
